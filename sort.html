function Sort()
{
var a=new Array();
var n=window.prompt("enter a limit");
for(var i=0;i< n;i++)
a[i]=parseInt(window.prompt("enter"+i+"th element"));
for( var i=0;i< n;i++)
document.write(a[i]+'< br/>');
for( var i=0;i< n;i++)
{
for(j=i+1;j< n;j++)
{
if(a[i]>a[j])
{
var t=a[i];
a[i]=a[j];
a[j]=t;
}
}
}
document.write('< br/>'+"sorted elements"+'< br/>');
for( var i=0;i< n;i++)
document.write(a[i]+'< br/>');
}
