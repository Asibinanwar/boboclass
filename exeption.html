import java.util.*;
class Maths
{
double findQuotient(int num1, int num2) throws ArithmeticException,
NumberFormatException,InputMismatchException
{
double result = num1/num2;
return result;
}
}
class ExceptionHandle
{
public static void main(String args[])
{
Scanner sc = new Scanner(System.in);
System.out.println("Enter 2 Numbers :- ");
try
{
int num1 = sc.nextInt();
int num2 = sc.nextInt();
Maths m = new Maths();
System.out.println(num1+" / "+ num2 + " = "+m.findQuotient(num1, num2));
}
catch(ArithmeticException e)
{
System.out.println("\t A Number cannot be divided by Zero...");
}
catch(NumberFormatException e)
{
System.out.println("\t Please do enter 2 integer values...");
}
catch(InputMismatchException e)
{
System.out.println("\t Two integers are expected....");
}
catch(Exception e)
{
System.out.println("\t Something went wrong....");
}
}
}
