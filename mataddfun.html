def read_matrix():

# read the number of rows and columns

rows = int(input("Enter the number of rows: "))

cols = int(input("Enter the number of columns: "))

# read the matrix elements

matrix = []

for i in range(rows):

row = []

for j in range(cols):

element = int(input(f"Enter the element at row {i+1}, column {j+1}: "))

row.append(element)

matrix.append(row)

return matrix

def add_matrices(matrix1, matrix2):

# check if the matrices have the same dimensions

if len(matrix1) != len(matrix2) or len(matrix1[0]) != len(matrix2[0]):

print("Error: matrices have different dimensions")

return None

# add the matrices element-wise

result = []

for i in range(len(matrix1)):

row = []

for j in range(len(matrix1[0])):

element = matrix1[i][j] + matrix2[i][j]

row.append(element)

result.append(row)

return result

# read the first matrix

print("Enter the first matrix:")

matrix1 = read_matrix()

# read the second matrix

print("Enter the second matrix:")

matrix2 = read_matrix()

# add the matrices and print the result

result = add_matrices(matrix1, matrix2)

if result is not None:

print("The sum of the matrices is:")

for row in result:

print(row)
