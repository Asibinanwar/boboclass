bookshop = {} # initialize an empty dictionary to store the books and their stock
while True:
print("Welcome to the bookshop!")
print("1. Add a book")
print("2. Sell a book")
print("3. Display all books")
print("4. Exit")
choice = input("Enter your choice: ")
if choice == "1":
# add a book
title = input("Enter the title of the book: ")
copies = int(input("Enter the number of copies to add: "))
if title in bookshop:
bookshop[title] += copies # add to existing stock
else:
bookshop[title] = copies # add new book and stock
print(f"{copies} copies of '{title}' added to the bookshop.")
elif choice == "2":
# sell a book
title = input("Enter the title of the book: ")
if title in bookshop and bookshop[title] > 0:
bookshop[title] -= 1 # reduce stock by 1
print(f"'{title}' sold. {bookshop[title]} copies left in stock.")
else:
print(f"'{title}' is not available or out of stock.")
elif choice == "3":
# display all books
if not bookshop:
print("The bookshop is currently empty.")
else:
print("Available books and their stock:")
for title, copies in bookshop.items():
print(f"'{title}': {copies} copies")
elif choice == "4":
# exit the program
print("Thank you for using the bookshop program.")
break
else:
print("Invalid choice. Please try again.")
