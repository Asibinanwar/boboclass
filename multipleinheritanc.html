class Student:

def __init__(self, name, roll):

self.name = name

self.roll = roll

def accept_details(self):

self.name = input("Enter student name: ")

self.roll = input("Enter student roll number: ")

def display_details(self):

print("Student Name:", self.name)

print("Student Roll Number:", self.roll)

class Sports:

def __init__(self):

self.sports_mark = 0

def accept_sports_mark(self):

self.sports_mark = int(input("Enter sports mark: "))

def display_sports_mark(self):

print("Sports Mark:", self.sports_mark)

class Statement(Student, Sports):

def __init__(self, name, roll):

Student.__init__(self, name, roll)

Sports.__init__(self)

self.total_marks = 0

def accept_subject_mark(self):

subject_mark = int(input("Enter subject mark: "))

self.total_marks += subject_mark

def display_total_marks(self):

print("Total Marks:", self.total_marks + self.sports_mark)

# create an object of the Statement class

student = Statement("John", "001")

# accept and display student and sports details

student.accept_details()

student.accept_sports_mark()

student.display_details()

student.display_sports_mark()

# accept and display subject marks and total marks

student.accept_subject_mark()

student.display_total_marks()
