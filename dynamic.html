class Animal
{
void makeSound()
{
System.out.println("My Sound is Unknown....");
}
}
class Cat extends Animal
{
void makeSound()
{
System.out.println("My Sound is MEOW MEOW....");
}
}
class Dog extends Animal
{
void makeSound()
{
System.out.println("My Sound is BOW BOW....");
}
}
class DynamicMethodDispatch
{
public static void main(String args[])
{
Animal ref = new Animal();
Cat c1 = new Cat();
Dog d1 = new Dog();
ref.makeSound();
ref = c1;
ref.makeSound();
ref = d1;
ref.makeSound();
}
}
