def reverse_digits(num):
# convert the number to a string and reverse it using slicing
reversed_str = str(num)[::-1]
# convert the reversed string back to an integer
reversed_num = int(reversed_str)
# count the number of digits in the original number
num_digits = len(str(num))
# return the reversed number and the number of digits
return reversed_num, num_digits
# read a number from the user
num = int(input("Enter a number: "))
# call the function to reverse the digits and count the number of digits
reversed_num, num_digits = reverse_digits(num)
# print the results
print(f"The reverse of {num} is {reversed_num}.")
print(f"{num} has {num_digits} digits.")
