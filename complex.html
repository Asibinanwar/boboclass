import java.util.*;
class Complex
{
int realPrt;
int imgPrt;
Complex()
{
realPrt = 0;
imgPrt = 0;
}
Complex(int realPrt, int imgPrt)
{
this.realPrt = realPrt;
this.imgPrt = imgPrt;
}
Complex addComplex(Complex c)
{
Complex res = new Complex();
res.realPrt = this.realPrt + c.realPrt;
res.imgPrt = this.imgPrt + c.imgPrt;
return res;
}
void dispComplex()
{
System.out.println(realPrt+"+ i"+imgPrt);
}
}
class ComplexSum
{
public static void main(String args[])
{
int a;
int b;
Scanner sc = new Scanner(System.in);
System.out.println("--- Complex 1 ---");
System.out.print("\tReal Part :- \t");
a = sc.nextInt();
System.out.print("\tImaginary Part :- \t");
b = sc.nextInt();
Complex c1 = new Complex(a,b);
System.out.println("--- Complex 2 ---");
System.out.print("\tReal Part :- \t");
a = sc.nextInt();
System.out.print("\tImaginary Part :- \t");
b = sc.nextInt();
Complex c2 = new Complex(a,b);
Complex Res = c1.addComplex(c2);
System.out.print("Complex 1 :- \t");
c1.dispComplex();
System.out.print("Complex 2 :- \t");
c2.dispComplex();
System.out.print("Sum Of 2 Complex :- \t");
Res.dispComplex();
}
}
