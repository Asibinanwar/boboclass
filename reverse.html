import java.util.*;
class RevArray
{
public static void main(String args[])
{
Scanner sc = new Scanner(System.in);
System.out.print("Enter the size of array :- \t");
int n = sc.nextInt();
int numsAry[] = new int[n];
int revAry[] = new int[n];
System.out.println("Enter the elements of array :- ");
for(int i=0; i< n; i++)
{
numsAry[i] = sc.nextInt();
}
for(int i=0; i< n; i++)
{
revAry[i] = numsAry[n-i-1];
}
System.out.println("Reverse array is :- ");
for(int i=0; i< n; i++)
{
System.out.print(revAry[i] + "\t");
}
System.out.println();
}
}
