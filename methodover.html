class Data
{
void addData(int x, int y)
{
System.out.println("Inputs are "+x+" and "+y);
int sum = x+y;
System.out.println("\tThe Result is "+sum);
}
void addData(double x, double y)
{
System.out.println("Inputs are "+x+" and "+y);
double sum = x + y;
System.out.println("\tThe Result is "+sum);
}
void addData(String x, String y)
{
System.out.println("Inputs are "+x+" and "+y);
String conCat = x+y;
System.out.println("\tThe Result is "+conCat);
}
}
class MethodOverload
{
public static void main(String args[])
{
Data d1 = new Data();
d1.addData(5,6);
d1.addData("Hello","World");
d1.addData(5.6,6.3);
}
}
