< script>
function findLargestAndFactorial() {
var numbers = document.getElementById("numbers").value.split(",");
var largest = parseInt(numbers[0]);
for (var i = 1; i < numbers.length; i++) {
var num = parseInt(numbers[i]);
if (num > largest) {
largest = num;
}
}
document.getElementById("largest").innerHTML = "Largest number: " + largest;
var factorial = 1;
for (var i = 1; i <= largest; i++) {
factorial *= i;
}
document.getElementById("factorial").innerHTML = "Factorial of " + largest + ": " + factorial;
}
< /script>
