#include< stdio.h>
void main()
{
int p[20],pr[20],bt[20],wt[20],tat[20],i,k,n,temp;
float wtavg,tatavg;
printf("\n Enter the number of process -- ");
scanf("%d",&n);
for(i=0;i< n;i++)
{
p[i]=i;
printf("Enter burst time and priority for process %d - ",i);
scanf("%d%d",&bt[i],&pr[i]);
}
for(i=0;i< n;i++)
for(k=i+1;k< n;k++)
if(pr[i]>pr[k])
{
temp=pr[i];
pr[i]=pr[k];
pr[k]=temp;
temp=p[i];
p[i]=p[k];
p[k]=temp;
}
wt[0]=wtavg=0;
tat[0]=tatavg=bt[0];
for(i=0;i< n;i++)
{
wt[i]=wt[i-1]+bt[i-1];
tat[i]=tat[i-1]+bt[i];
wtavg=wtavg+wt[i];
tatavg=tatavg+tat[i];
}
printf("\nPROCESS \t PRIORITY \t BRUST TIME \t WAITING TIME \t TURNAROUND TIME\n");
for(i=0;i< n;i++)
{
printf("\n %d \t\t %d \t\t %d \t\t %d \t\t %d",p[i],pr[i],bt[i],wt[i],tat[i]);
}
printf("\n Average waiting time - %f", wtavg/n);
printf("\n Average turnaround time - %f\n",tatavg/n);
}
