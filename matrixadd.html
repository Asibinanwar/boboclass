# read the dimensions of the matrices from the user
rows = int(input("Enter the number of rows: "))
cols = int(input("Enter the number of columns: "))
# create the first matrix and read its values from the user
print("Enter the values of the first matrix:")
matrix1 = []
for i in range(rows):
row = []
for j in range(cols):
value = int(input(f"Enter the value of element ({i}, {j}): "))
row.append(value)
matrix1.append(row)
# create the second matrix and read its values from the user
print("Enter the values of the second matrix:")
matrix2 = []
for i in range(rows):
row = []
for j in range(cols):
value = int(input(f"Enter the value of element ({i}, {j}): "))
row.append(value)
matrix2.append(row)
# add the matrices element-wise and create the result matrix
result_matrix = []
for i in range(rows):
row = []
for j in range(cols):
sum = matrix1[i][j] + matrix2[i][j]
row.append(sum)
result_matrix.append(row)
# print the result matrix
print("The result of matrix addition is:")
for row in result_matrix:
print(row)
