import java.util.*;
class SparseMatrixException extends Exception
{
SparseMatrixException()
{
super();
}
}
class UserDefinedExceptionDemo
{
public static void main(String args[])
{
int mat[][] = new int[3][3];
Scanner sc = new Scanner(System.in);
int numZeros = 0;
System.out.println("Enter the elements of the Matrix :- ");
for(int i=0; i< 3; i++)
{
for(int j=0; j< 3; j++)
{
mat[i][j] = sc.nextInt();
if(mat[i][j] == 0)
numZeros++;
}
}
try
{
if(numZeros > 5)
throw new SparseMatrixException();
else
{
System.out.println("The given matrix is :- ");
for(int i=0; i< 3; i++)
{
for(int j=0; j< 3; j++)
{
System.out.print("\t" + mat[i][j]);
}
System.out.println();
}
}
}
catch(SparseMatrixException e)
{
System.out.println("The given matrix is a sparse matrix. Please enter non sparse matrix.");
}
}
}
