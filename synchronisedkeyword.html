class Cricket
{
static synchronized void dispPlayers(String s1, String s2)
{
try
{
Thread.sleep(1000);
System.out.print(" "+s1);
System.out.print(" "+s2);
System.out.println();
}
catch(InterruptedException e)
{
System.out.println(e);
}
}
}
class Thread1 extends Thread
{
public void run()
{
Cricket.dispPlayers("Virat","Kohli");
}
}
class Thread2 extends Thread
{
public void run()
{
Cricket.dispPlayers("Babar", "Azam");
}
}
class SynchroDemo
{
public static void main(String args[])
{
Thread1 t1 = new Thread1();
Thread2 t2 = new Thread2();
t1.start();
t2.start();
}
}
