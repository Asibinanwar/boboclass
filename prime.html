function prime()
{
var n=window.prompt("enter the limit:");
var a=new Array();
var k=0;
for(var i=2;i<=n;i++)
{
let flag=0;
for(let j=2;j< i;j++)
{
if(i%j==0)
{
flag=1;
}
}
if(flag==0)
{
a[k]=i;
k++;
}
}
document.write("prime numbers are"+'< br>')
for(var i=0;i< k;i++)
{
document.write(a[i]+'< br>')
}
}
prime()
