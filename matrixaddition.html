import java.util.*;
class Matrix
{
int row;
int col;
int a[][];
Matrix(int row, int col)
{
this.row = row;
this.col = col;
a = new int[row][col];
}
void readMatrix()
{
Scanner sc = new Scanner(System.in);
System.out.println("Enter elements of the matrix :- ");
for(int i=0; i< row; i++)
{
for(int j=0; j< col; j++)
{
a[i][j] = sc.nextInt();
}
}
}
Matrix addMatrix(Matrix m)
{
Matrix res = new Matrix(row,col);
for(int i=0; i< row; i++)
{
for(int j=0; j< col; j++)
{
res.a[i][j] = this.a[i][j] + m.a[i][j];
}
}
return res;
}
void dispMatrix()
{
for(int i=0; i< row; i++)
{
for(int j=0; j< col; j++)
{
System.out.print(a[i][j]+"\t");
}
System.out.println();
}
}
}
class MatAdd
{
public static void main(String args[])
{
Matrix m1 = new Matrix(3,3);
Matrix m2 = new Matrix(3,3);
m1.readMatrix();
m2.readMatrix();
Matrix res = m1.addMatrix(m2);
System.out.println("\n---- Matrix M1 ----");
m1.dispMatrix();
System.out.println();
System.out.println("---- Matrix M2 ----");
m2.dispMatrix();
System.out.println();
System.out.println("---- Resultant Matrix ----");
res.dispMatrix();
System.out.println();
}
}
